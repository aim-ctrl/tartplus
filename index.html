Klicka på ett segment

import js from pyodide.ffi import create_proxy # --- DATASTRUKTUR --- # Data över vilka kapitel som ligger i varje Juz, samt hur många verser (v) de utgör av just den Juz:en. juz_data = [ # Juz 1: Al-Fatihah (7 verser) + Al-Baqarah (141 verser) = 148 totalt [{"id": 1, "name": "1. Al-Fatihah", "v": 7}, {"id": 2, "name": "2. Al-Baqarah", "v": 141}], # Juz 2: Fortsättning på Al-Baqarah (111 verser) [{"id": 2, "name": "2. Al-Baqarah", "v": 111}], # Juz 3: Slutet på Al-Baqarah (34 verser) + Aal-E-Imran (92 verser) [{"id": 2, "name": "2. Al-Baqarah", "v": 34}, {"id": 3, "name": "3. Aal-E-Imran", "v": 92}], ] # Fyller ut Juz 4 till 30 med platshållardata för att visualiseringen ska fungera direkt for i in range(4, 31): juz_data.append([{"id": i, "name": f"Surah {i}", "v": 100}]) # Databas med statistik för kapitlen (som visas vid klick) surah_stats = { 1: {"name": "Al-Fatihah", "meaning": "Öppningen", "total_verses": 7, "revelation": "Mekka"}, 2: {"name": "Al-Baqarah", "meaning": "Kon", "total_verses": 286, "revelation": "Medina"}, 3: {"name": "Aal-E-Imran", "meaning": "Imrans Ätt", "total_verses": 200, "revelation": "Medina"} } # Färger för att särskilja kapitlen colors = ["#4A90E2", "#50E3C2", "#F5A623", "#D0021B", "#8B572A", "#9B59B6", "#34495E", "#16A085", "#27AE60", "#2980B9"] # --- LOGIK --- # Funktion som körs när användaren klickar på ett segment def show_stats(event): # Hämta ID från det klickade HTML-elementet surah_id = int(event.target.getAttribute("data-id")) # Hämta statistik, eller visa standardtext om det saknas i vår demo-databas stats = surah_stats.get(surah_id, { "name": f"Kapitel {surah_id}", "meaning": "Ej inlagd", "total_verses": "Okänt", "revelation": "Okänt" }) # Uppdatera HTML i statistiken js.document.getElementById("surah-title").innerText = f"{surah_id}. {stats['name']}" js.document.getElementById("surah-info").innerHTML = ( f"Betydelse: {stats['meaning']}
" f"Totalt antal verser (hela kapitlet): {stats['total_verses']}
" f"Uppenbarelseplats: {stats['revelation']}" ) # Gör panelen synlig js.document.getElementById("stats-panel").style.display = "block" # PyScript behöver en proxy för att koppla Python-funktioner till JavaScript-händelser click_proxy = create_proxy(show_stats) # Bygg DOM (HTML-elementen) för alla 30 Juz container = js.document.getElementById("quran-container") for juz_index, juz in enumerate(juz_data): # Räkna ut totala antalet verser i denna Juz för att kunna beräkna procent total_verses_in_juz = sum(segment["v"] for segment in juz) # Skapa en rad (Rektangeln för en Juz) row = js.document.createElement("div") row.className = "juz-row" row.title = f"Juz {juz_index + 1}" # Gå igenom varje kapitel-segment i denna Juz for surah in juz: segment = js.document.createElement("div") segment.className = "surah-segment" # Beräkna bredden i procent baserat på antalet verser percentage_width = (surah["v"] / total_verses_in_juz) * 100 segment.style.width = f"{percentage_width}%" # Sätt färg baserat på ID så att samma kapitel får samma färg oavsett Juz color_index = surah["id"] % len(colors) segment.style.backgroundColor = colors[color_index] # Lägg till text och attribut segment.innerText = surah["name"] segment.setAttribute("data-id", str(surah["id"])) # Koppla klick-händelsen segment.addEventListener("click", click_proxy) # Lägg in segmentet i Juz-raden row.appendChild(segment) # Lägg in hela Juz-raden i vår huvudbehållare container.appendChild(row)
